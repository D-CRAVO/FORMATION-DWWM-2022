EXERCICE 3.3 INTERVALLE ENTRE DEUX NOMBRES

# VARIABLES

ENTIER nombre_1
ENTIER nombre_2
ENTIER temporaire
ENTIER resultat

# DEBUT PROGRAMME

ECRIRE "Veuillez saisir le premier nombre entier : "
LIRE nombre_1
ECRIRE "Veuillez saisir le deuxième nombre entier : "
LIRE nombre_2
ECRIRE "Résultat = "

# SOLUTION 1

SI nombre_1 < nombre_2
ALORS
    resultat <-- nombre_1
    POUR 
       i ALLANT DE (nombre_1 + 1) A (nombre_2 - 1)
    FAIRE 
       resultat <-- resultat + 1
       ECRIRE resultat
    FIN POUR
SINON 
       resultat <-- nombre_2
   POUR 
       i ALLANT DE (nombre_1 - 1) A (nombre_2 + 1) 
   FAIRE
       resultat <-- resultat - 1
       ECRIRE resultat
   FIN POUR
FIN SI

# SOLUTION 2

SI nombre_1 > nombre_2
ALORS
    temporaire <-- nombre_2
    nombre_2 <-- nombre_1
    nombre_1 <-- temporaire
FIN SI

resultat <-- nombre_1
POUR 
    i ALLANT DE (nombre_1 + 1) A (nombre_2 - 1)
FAIRE 
    resultat <-- resultat + 1
FIN POUR
ECRIRE resultat

# FIN PROGRAMME
